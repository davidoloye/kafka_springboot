#Create Topic
docker exec broker kafka-topics --bootstrap-server broker:9092 --create --topic {topicName}

#List Topic
docker exec broker kafka-topics --bootstrap-server broker:9092 --list

#Producer Write to Topic
docker exec --interactive --tty broker kafka-console-producer --bootstrap-server broker:9092 --topic mytopic


#Consumer Read/Listen to Topic
docker exec --interactive --tty broker kafka-console-consumer --bootstrap-server broker:9092 --topic kidsTopic --from-beginning
